class FiixError(Exception):
    """Fiix API error with code, message, and stack trace."""
    def __init__(self, message="", code=None, leg=None, stack_trace=None):
        self.code = code
        self.leg = leg
        self.stack_trace = stack_trace
        super().__init__(f"[{code or 'ERR'}] {message}")

    # ---------------------------------------------------------------
    # INITIALIZATION
    # ---------------------------------------------------------------

    def __init__(self, subdomain: str, app_key: str, access_key: str,
                 secret_key: str, timeout: int = 30, max_retries: int = 3):
        self.subdomain = subdomain
        self.app_key = app_key
        self.access_key = access_key
        self.secret_key = secret_key
        self.timeout = timeout
        self.base_url = f"https://{subdomain}.macmms.com/api/"
        self.upload_url = f"https://{subdomain}.macmms.com/api/upload"
        self.client_version = {"major": 2, "minor": 48, "patch": 1}
        self._request_count = 0

        # Build HTTP session with retries
        self._session = requests.Session()
        retry = Retry(total=max_retries, backoff_factor=0.5,
                      status_forcelist=[429, 500, 502, 503, 504],
                      allowed_methods=["POST"])
        self._session.mount("https://", HTTPAdapter(max_retries=retry))

    # ---------------------------------------------------------------
    # AUTHENTICATION
    # ---------------------------------------------------------------

    def _build_url(self, base_url=None, extra_params=None):
        """Build URL with auth query parameters."""
        params = {
            "service": "cmms",
            "timestamp": str(int(time.time() * 1000)),
            "appKey": self.app_key,
            "accessKey": self.access_key,
            "signatureMethod": "HmacSHA256",
            "signatureVersion": "1",
        }
        if extra_params:
            params.update(extra_params)
        return f"{base_url or self.base_url}?{urllib.parse.urlencode(params)}"

    def _sign(self, url):
        """Generate HMAC-SHA256 signature."""
        message = url
        for prefix in ("https://", "http://"):
            if message.startswith(prefix):
                message = message[len(prefix):]
                break
        return hmac.HMAC(
            key=self.secret_key.encode("utf-8"),
            msg=message.encode("utf-8"),
            digestmod=hashlib.sha256
        ).hexdigest().lower()

    # ---------------------------------------------------------------
    # HTTP POST
    # ---------------------------------------------------------------

    def _post(self, payload, base_url=None, extra_params=None):
        """Send authenticated POST request."""
        url = self._build_url(base_url, extra_params)
        signature = self._sign(url)
        headers = {
            "Content-Type": "text/plain",
            "Accept": "application/json",
            "Authorization": signature,
        }
        body = json.dumps(payload, separators=(",", ":"))

        logger.debug(f"POST {url[:120]}...")
        try:
            resp = self._session.post(url, data=body.encode("utf-8"),
                                       headers=headers, timeout=self.timeout)
        except requests.exceptions.ConnectionError as e:
            raise FiixError(f"Connection failed: {e}")
        except requests.exceptions.Timeout:
            raise FiixError("Request timed out")

        self._request_count += 1

        # Check HTTP-level errors (rare — Fiix returns 200 even on API errors)
        if resp.status_code != 200:
            raise FiixError(f"HTTP {resp.status_code}: {resp.text[:300]}")

        try:
            result = resp.json()
        except json.JSONDecodeError:
            raise FiixError(f"Invalid JSON response: {resp.text[:200]}")

        # Fiix API returns errors inside the response body, not via HTTP status
        if result.get("error"):
            err = result["error"]
            raise FiixError(err.get("message", "Unknown"), err.get("code"),
                           err.get("leg"), err.get("stackTrace"))
        return result

    def _base_payload(self):
        return {
            "clientVersion": self.client_version,
            "requestSentUnixTime": int(time.time() * 1000),
        }

    # ---------------------------------------------------------------
    # CRUD OPERATIONS
    # ---------------------------------------------------------------

    def find(self, class_name, fields="id", filters=None,
             order_by=None, start_index=0, max_objects=100):
        """Find/query objects (CRUD Read)."""
        payload = self._base_payload()
        payload.update({
            "_maCn": "FindRequest", "className": class_name,
            "fields": fields, "startIndex": start_index,
            "maxObjects": min(max_objects, 1000),
        })
        if filters:
            payload["filters"] = filters
        if order_by:
            payload["orderBy"] = order_by
        return self._post(payload)

    def find_by_id(self, class_name, obj_id, fields="id"):
        """Find a single object by ID."""
        result = self.find(class_name, fields,
                          filters=[{"ql": "id = ?", "parameters": [obj_id]}],
                          max_objects=1)
        objs = result.get("objects", [])
        return objs[0] if objs else None

    def find_all(self, class_name, fields="id", filters=None,
                 order_by=None, page_size=500, max_pages=200):
        """Auto-paginate to fetch ALL records. max_pages caps at 100k records."""
        all_objects = []
        start = 0
        page = 0
        while page < max_pages:
            result = self.find(class_name, fields, filters, order_by, start, page_size)
            objs = result.get("objects", [])
            if not objs:
                break
            all_objects.extend(objs)
            if len(objs) < page_size:
                break
            start += page_size
            page += 1
            if page % 10 == 0:
                logger.info(f"  Paginating {class_name}: {len(all_objects):,} records (page {page})...")
        return all_objects

    def add(self, class_name, object_data, fields="id"):
        """Create a new object (CRUD Create)."""
        payload = self._base_payload()
        payload.update({
            "_maCn": "AddRequest", "className": class_name,
            "object": {**object_data, "className": class_name},
            "fields": fields,
        })
        return self._post(payload)

    def change(self, class_name, object_data, change_fields="", fields="id"):
        """Update an existing object (CRUD Update)."""
        if not change_fields:
            change_fields = ",".join(k for k in object_data if k not in ("id", "className"))
        payload = self._base_payload()
        payload.update({
            "_maCn": "ChangeRequest", "className": class_name,
            "object": {**object_data, "className": class_name},
            "fields": fields, "changeFields": change_fields,
        })
        return self._post(payload)

    def remove(self, class_name, obj_id):
        """Delete an object by ID (CRUD Delete)."""
        payload = self._base_payload()
        payload.update({"_maCn": "RemoveRequest", "className": class_name, "id": obj_id})
        return self._post(payload)

    def batch(self, requests_list):
        """Execute multiple requests in one batch call."""
        payload = self._base_payload()
        payload.update({"_maCn": "BatchRequest", "requests": requests_list})
        return self._post(payload)

    def upload_file(self, descriptions, files):
        """Upload files to Fiix."""
        url = self._build_url(self.upload_url, {"action": "uploadFile"})
        signature = self._sign(url)
        headers = {"Authorization": signature}
        multipart = [("data", (None, json.dumps(descriptions), "application/json"))]
        for f in files:
            multipart.append(("file", f))
        resp = self._session.post(url, files=multipart, headers=headers,
                                   timeout=self.timeout * 3)
        self._request_count += 1
        return resp.json()

    # ---------------------------------------------------------------
    # INTERNAL HELPERS
    # ---------------------------------------------------------------

    def _fetch_all(self, class_name, fields, filters=None, max_objects=500):
        """Internal: fetch all records for a class with pagination."""
        return self.find_all(class_name, fields, filters, page_size=max_objects)

    def _rpc(self, name, action, params=None):
        """Internal: execute an RPC call."""
        payload = self._base_payload()
        payload.update({"_maCn": "RpcRequest", "name": name, "action": action})
        if params:
            payload["parameters"] = params
        return self._post(payload)

    # ---------------------------------------------------------------
    # CONNECTION TEST
    # ---------------------------------------------------------------

    def test_connection(self):
        """Test connectivity. Returns True on success."""
        try:
            result = self.rpc_ping()
            return "error" not in result or not result.get("error")
        except Exception:
            return False

    @property
    def request_count(self):
        """Total API calls made in this session."""
        return self._request_count
